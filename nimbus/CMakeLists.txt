 setup_boost(program_options filesystem python regex)
 setup_qt()

 find_package(HURRICANE          REQUIRED)
 find_package(CORIOLIS           REQUIRED)
 if(BUILD_DOC)
   find_package(Doxygen)
 endif(BUILD_DOC)
 
 ADD_SUBDIRECTORY(src)
